🏗️ Data Structure Definition

The core data structure is a struct that holds all information about an emergency request, and a Priority Queue struct that holds an array-based Max-Heap.

main.c — Data Structures C 
/* Represents a single emergency request */
typedef struct {
    int    id;
    char   location[50];
    char   disaster_type[30];   /* flood, earthquake, fire, accident */
    int    people_affected;
    int    urgency;             /* 1 (low) to 10 (critical) */
    char   resources_needed[50]; /* food, medical, rescue */
} EmergencyRequest;

/* Max-Heap Priority Queue */
typedef struct {
    EmergencyRequest data[MAX_QUEUE_SIZE];
    int size;
} PriorityQueue;

⬆️ HeapifyUp — Insert Operation

When a new request is inserted, it is placed at the end of the array. The heapifyUp function then compares it with its parent and swaps upward until the heap property is restored.

main.c — heapifyUp() C 
/* Restore heap property upward after insertion */
void heapifyUp(PriorityQueue *pq, int index) {
    while (index > 0) {
        int parent = (index - 1) / 2;
        if (pq->data[index].urgency > pq->data[parent].urgency) {
            swap(&pq->data[index], &pq->data[parent]);
            index = parent;
        } else {
            break;
        }
    }
}

⬇️ HeapifyDown — Extract Operation

When the root (highest priority) is extracted, the last element takes its place. The heapifyDown function pushes it down by comparing with children until the heap property is restored.

main.c — heapifyDown() C 
/* Restore heap property downward after extraction */
void heapifyDown(PriorityQueue *pq, int index) {
    int largest = index;
    int left    = 2 * index + 1;
    int right   = 2 * index + 2;

    if (left < pq->size && pq->data[left].urgency > pq->data[largest].urgency)
        largest = left;
    if (right < pq->size && pq->data[right].urgency > pq->data[largest].urgency)
        largest = right;

    if (largest != index) {
        swap(&pq->data[index], &pq->data[largest]);
        heapifyDown(pq, largest);
    }
}

📥 Insert & 📤 ExtractMax

main.c — insert() C 
/* Insert a new emergency request into the priority queue */
int insert(PriorityQueue *pq, EmergencyRequest req) {
    if (pq->size >= MAX_QUEUE_SIZE) {
        printf("\n  [ERROR] Priority Queue is full!\n");
        return 0;
    }
    pq->data[pq->size] = req;
    heapifyUp(pq, pq->size);
    pq->size++;
    return 1;
}
main.c — extractMax() C 
/* Extract the highest-priority (max urgency) request */
int extractMax(PriorityQueue *pq, EmergencyRequest *result) {
    if (pq->size <= 0) {
        printf("\n  [INFO] No pending requests.\n");
        return 0;
    }
    *result = pq->data[0];
    pq->data[0] = pq->data[pq->size - 1];
    pq->size--;
    if (pq->size > 0)
        heapifyDown(pq, 0);
    return 1;
}

🚀 Main Function & Menu

The main function initializes the queue and presents a menu-driven interface allowing the user to add requests, allocate resources, view pending requests, check the next priority, view history, or exit.

main.c — main() C 
int main() {
    int choice;
    initQueue(&queue);

    do {
        displayMenu();
        scanf("%d", &choice);

        switch (choice) {
            case 1: addRequest();        break;
            case 2: allocateResources(); break;
            case 3: viewAllRequests();   break;
            case 4: viewNextPriority();  break;
            case 5: viewHistory();       break;
            case 6: /* Exit */          break;
            default: printf("Invalid choice.\n");
        }
    } while (choice != 6);

    return 0;
}
